    <script>
        // ⚠️ ЗАМЕНИТЕ ЭТОТ URL на URL вашего развернутого веб-приложения Google Apps Script
        const GOOGLE_SCRIPT_URL = 'https://docs.google.com/spreadsheets/d/1lYYgZTHQaPTuHO-tRwqmE6KztLnHIk4BOVv04nshkxE/edit?pli=1&gid=0#gid=0c';
        
        document.getElementById('dataForm').addEventListener('submit', async function(e) {
            e.preventDefault();
            
            const submitBtn = document.getElementById('submitBtn');
            const loading = document.getElementById('loading');
            const status = document.getElementById('status');
            
            // Проверка URL
            if (GOOGLE_SCRIPT_URL === 'ВАШ_URL_ВЕБ_ПРИЛОЖЕНИЯ_ЗДЕСЬ') {
                status.className = 'status error';
                status.textContent = '❌ Ошибка: Укажите URL веб-приложения в коде (строка 245)';
                status.style.display = 'block';
                return;
            }
            
            // Собираем данные из формы
            const formData = {
                field1: document.getElementById('field1').value,
                field2: document.getElementById('field2').value,
                field3: document.getElementById('field3').value,
                field4: document.getElementById('field4').value,
                field5: document.getElementById('field5').value,
                field6: document.getElementById('field6').value,
                field7: document.getElementById('field7').value,
                field8: document.getElementById('field8').value,
                field9: document.getElementById('field9').value,
                field10: document.getElementById('field10').value,
                field11: document.getElementById('field11').value,
                field12: document.getElementById('field12').value,
                field13: document.getElementById('field13').value
            };
            
            // UI состояние загрузки
            submitBtn.disabled = true;
            loading.style.display = 'block';
            status.style.display = 'none';
            
            try {
                const response = await fetch(GOOGLE_SCRIPT_URL, {
                    method: 'POST',
                    headers: {
                        'Content-Type': 'application/json',
                    },
                    body: JSON.stringify(formData)
                });
                
                const result = await response.json();
                
                if (result.status === 'success') {
                    status.className = 'status success';
                    status.textContent = '✅ ' + result.message;
                    document.getElementById('dataForm').reset();
                } else {
                    throw new Error(result.message);
                }
                
            } catch (error) {
                status.className = 'status error';
                status.textContent = '❌ Ошибка: ' + error.message;
                console.error('Ошибка:', error);
            } finally {
                submitBtn.disabled = false;
                loading.style.display = 'none';
                status.style.display = 'block';
                
                // Скрыть статус через 5 секунд
                setTimeout(() => {
                    status.style.display = 'none';
                }, 5000);
            }
        });
    </script>
